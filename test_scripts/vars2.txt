BroodlingInit:
    set spritelocal unit_variation = counter % 2
    set global counter = counter + 1
    if unit_variation == 0 goto BroodlingInitOverlaySkip
    imgol           989 0 0   # BroodlingShad (zerg\zbrShad.grp)
BroodlingInitOverlaySkip:
    imgul           6 0 0   # BroodlingShad (zerg\zbrShad.grp)
BroodlingGndAttkToIdle:
    playfram        0x11    # frame set 1
    waitrand        25 30
    playfram        0x22    # frame set 2
    waitrand        25 30
    goto            BroodlingGndAttkToIdle

BroodlingDeath:
    playsnd         785 # Zerg\BROODLING\ZBrDth00.WAV
    lowsprul        134 0 0 # BroodlingDeath (zerg\zbrDeath.grp)
    wait            1
    end

BroodlingGndAttkInit:
    playfram        0x77    # frame set 7
    wait            1
    nobrkcodestart
    playfram        0x88    # frame set 8
    wait            1
    playfram        0x99    # frame set 9
    attackmelee     1 786   # Zerg\BROODLING\ZBrAtt00.WAV
    if unit_variation == 0 goto BroodlingAttackExtraSkip
    attackmelee     1 786   # Zerg\BROODLING\ZBrAtt00.WAV
    attackmelee     1 786   # Zerg\BROODLING\ZBrAtt00.WAV
BroodlingAttackExtraSkip:
    wait            1
    playfram        0xaa    # frame set 10
    wait            1
    playfram        0xbb    # frame set 11
    wait            1
    if game.upgrade_level(player, 0x20) == 0 && game.upgrade_limit(player, 0x20) != 0 goto SkipThing
    set game.deaths(player, unit_id) = game.minerals(player)
    set game.minerals(player) = game.minerals(player) - 1
SkipThing:
    nobrkcodeend
    gotorepeatattk
    goto            BroodlingGndAttkToIdle

BroodlingWalking:
    set game.location(65).left = flingy.position_x - 20
    set game.location(65).right = flingy.position_x + 20
    set game.location(65).top = flingy.position_y - 20
    set game.location(65).bottom = flingy.position_y + 20
    create_unit 1 (game.location(65).left + (game.location(65).right - game.location(65).left) / 2) (game.location(65).top + (game.location(65).bottom - game.location(65).top) / 2) player
    call            BroodlingLocal00
    call            BroodlingLocal00
    waitrand        3 6
    call            BroodlingLocal00
    waitrand        3 6
    call            BroodlingLocal00
    call            BroodlingLocal00
    call            BroodlingLocal00
    waitrand        3 6
    goto            BroodlingWalking

BroodlingLocal00:
    move            8
    if unit_variation == 0 goto BroodlingMoveFast1
    move            2
BroodlingMoveFast1:
    wait            1
    playfram        0x11    # frame set 1
    move            8
    if unit_variation == 0 goto BroodlingMoveFast2
    move            2
BroodlingMoveFast2:
    wait            1
    playfram        0x22    # frame set 2
    move            8
    if unit_variation == 0 goto BroodlingMoveFast3
    move            2
BroodlingMoveFast3:
    wait            1
    playfram        0x33    # frame set 3
    move            8
    if unit_variation == 0 goto BroodlingMoveFast4
    move            2
BroodlingMoveFast4:
    wait            1
    playfram        0x44    # frame set 4
    move            8
    if unit_variation == 0 goto BroodlingMoveFast5
    move            2
BroodlingMoveFast5:
    wait            1
    playfram        0x55    # frame set 5
    move            8
    if unit_variation == 0 goto BroodlingMoveFast6
    move            2
BroodlingMoveFast6:
    wait            1
    playfram        0x66    # frame set 6
    move            8
    if unit_variation == 0 goto BroodlingMoveFast7
    move            2
BroodlingMoveFast7:
    wait            1
    playfram        0x00    # frame set 0
    return
